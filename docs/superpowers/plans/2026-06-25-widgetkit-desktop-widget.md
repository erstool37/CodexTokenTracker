# WidgetKit Desktop Widget Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a real macOS WidgetKit desktop widget for CodexTokenTracker, with the large widget showing Codex `/status`-style rate limits plus exact `/usage` daily, weekly, and cumulative account-history token totals.

**Architecture:** Keep the existing SwiftPM menu bar app and tests intact. Add shared snapshot models/stores in `CodexTokenTrackerCore`, have the menu bar app write display-safe widget snapshots after refresh, and add an XcodeGen-generated Xcode project with a macOS app target plus embedded WidgetKit extension target.

**Tech Stack:** Swift 6, SwiftUI, AppKit menu bar app, WidgetKit, Xcode 26.5 via `DEVELOPER_DIR=/Applications/Xcode.app/Contents/Developer`, XcodeGen for project generation.

```swift
import Foundation

public final class WidgetSnapshotStore: @unchecked Sendable {
    public static let appGroupIdentifier = "group.com.erstool37.CodexTokenTracker"
    private let urls: [URL]

    public init(urls: [URL] = WidgetSnapshotStore.defaultSnapshotURLs()) {
        self.urls = urls
    }

    public static func defaultSnapshotURLs(fileManager: FileManager = .default) -> [URL] {
        var candidates: [URL] = []
        if let groupURL = fileManager.containerURL(forSecurityApplicationGroupIdentifier: appGroupIdentifier) {
            candidates.append(groupURL.appendingPathComponent("widget-snapshot.json"))
        }
        let applicationSupport = fileManager.urls(for: .applicationSupportDirectory, in: .userDomainMask).first
            ?? fileManager.homeDirectoryForCurrentUser.appendingPathComponent("Library/Application Support", isDirectory: true)
        candidates.append(
            applicationSupport
                .appendingPathComponent("CodexTokenTracker", isDirectory: true)
                .appendingPathComponent("widget-snapshot.json")
        )
        return candidates
    }

    public func write(_ snapshot: WidgetSnapshot) throws {
        let encoder = JSONEncoder()
        encoder.dateEncodingStrategy = .iso8601
        encoder.outputFormatting = [.prettyPrinted, .sortedKeys]
        let data = try encoder.encode(snapshot)
        for url in urls {
            try write(data, to: url)
        }
    }

    public func read() throws -> WidgetSnapshot? {
        let decoder = JSONDecoder()
        decoder.dateDecodingStrategy = .iso8601
        for url in urls {
            guard FileManager.default.fileExists(atPath: url.path) else {
                continue
            }
            let data = try Data(contentsOf: url)
            return try decoder.decode(WidgetSnapshot.self, from: data)
        }
        return nil
    }

    private func write(_ data: Data, to url: URL) throws {
        try FileManager.default.createDirectory(
            at: url.deletingLastPathComponent(),
            withIntermediateDirectories: true
        )
        try data.write(to: url, options: .atomic)
    }
}
```

import CodexTokenTrackerCore
import SwiftUI
import WidgetKit

struct CodexTokenTrackerWidgetEntry: TimelineEntry {
    let date: Date
    let snapshot: WidgetSnapshot?
}

struct CodexTokenTrackerWidgetProvider: TimelineProvider {
    func placeholder(in context: Context) -> CodexTokenTrackerWidgetEntry {
        CodexTokenTrackerWidgetEntry(date: Date(), snapshot: WidgetPreviewData.placeholder)
    }

    func getSnapshot(in context: Context, completion: @escaping (CodexTokenTrackerWidgetEntry) -> Void) {
        completion(CodexTokenTrackerWidgetEntry(date: Date(), snapshot: loadSnapshot()))
    }

    func getTimeline(in context: Context, completion: @escaping (Timeline<CodexTokenTrackerWidgetEntry>) -> Void) {
        let now = Date()
        let entry = CodexTokenTrackerWidgetEntry(date: now, snapshot: loadSnapshot())
        let nextRefresh = Calendar.current.date(byAdding: .minute, value: 15, to: now) ?? now.addingTimeInterval(900)
        completion(Timeline(entries: [entry], policy: .after(nextRefresh)))
    }

    private func loadSnapshot() -> WidgetSnapshot? {
        try? WidgetSnapshotStore().read()
    }
}

struct CodexTokenTrackerWidgetView: View {
    @Environment(\.widgetFamily) private var family
    let entry: CodexTokenTrackerWidgetEntry

    var body: some View {
        Group {
            if let snapshot = entry.snapshot {
                content(snapshot)
            } else {
                unavailable
            }
        }
        .containerBackground(.background, for: .widget)
    }

    private func content(_ snapshot: WidgetSnapshot) -> some View {
        VStack(alignment: .leading, spacing: 8) {
            header(snapshot)
            ForEach(snapshot.limitBuckets, id: \.label) { bucket in
                limitBucket(bucket)
            }
            usage(snapshot)
            if let errorText = snapshot.errorText {
                Text(errorText)
                    .font(.caption2)
                    .foregroundStyle(.orange)
                    .lineLimit(2)
            }
            Spacer(minLength: 0)
            footer(snapshot)
        }
        .padding(14)
    }

    private func header(_ snapshot: WidgetSnapshot) -> some View {
        HStack {
            Text(snapshot.title)
                .font(.headline.weight(.semibold))
            Spacer()
            if entry.date > snapshot.staleAfter {
                Text("Stale")
                    .font(.caption2.weight(.semibold))
                    .foregroundStyle(.orange)
            }
        }
    }

    private func limitBucket(_ bucket: WidgetLimitBucket) -> some View {
        VStack(alignment: .leading, spacing: 4) {
            Text(bucket.label)
                .font(.caption.weight(.semibold))
            ForEach(bucket.windows, id: \.label) { window in
                HStack(alignment: .firstTextBaseline) {
                    Text(window.label)
                        .foregroundStyle(.secondary)
                    Spacer()
                    VStack(alignment: .trailing, spacing: 1) {
                        Text(window.percentLeftText)
                            .fontWeight(.semibold)
                            .foregroundStyle(color(for: window.warningLevel))
                        if let resetsAtText = window.resetsAtText {
                            Text("resets \(resetsAtText)")
                                .font(.caption2)
                                .foregroundStyle(.secondary)
                        }
                    }
                    .monospacedDigit()
                }
                .font(.caption2)
            }
            if let creditsText = bucket.creditsText {
                metricRow("Credits", creditsText)
            }
            if let statusText = bucket.statusText {
                metricRow("Status", statusText)
            }
        }
    }

    private func usage(_ snapshot: WidgetSnapshot) -> some View {
        VStack(alignment: .leading, spacing: 4) {
            Text("Usage")
                .font(.caption.weight(.semibold))
            ForEach(snapshot.usageRows, id: \.label) { row in
                metricRow(row.label, row.value)
            }
            if let note = snapshot.note {
                Text(note)
                    .font(.caption2)
                    .foregroundStyle(.secondary)
            }
        }
    }

    private func metricRow(_ label: String, _ value: String) -> some View {
        HStack(alignment: .firstTextBaseline) {
            Text(label)
                .foregroundStyle(.secondary)
            Spacer()
            Text(value)
                .fontWeight(.semibold)
                .monospacedDigit()
        }
        .font(.caption2)
    }

    private func footer(_ snapshot: WidgetSnapshot) -> some View {
        HStack {
            if let sourceText = snapshot.sourceText {
                Text(sourceText)
            }
            Spacer()
            Text(snapshot.refreshedAt, style: .time)
                .monospacedDigit()
        }
        .font(.caption2)
        .foregroundStyle(.secondary)
    }

    private var unavailable: some View {
        VStack(alignment: .leading, spacing: 6) {
            Text("Codex")
                .font(.headline.weight(.semibold))
            Text("Open CodexTokenTracker")
                .font(.caption)
                .foregroundStyle(.secondary)
        }
        .padding(14)
        .containerBackground(.background, for: .widget)
    }

    private func color(for warningLevel: LimitWarningLevel) -> Color {
        switch warningLevel {
        case .normal:
            return .primary
        case .warning:
            return .orange
        case .critical:
            return .red
        }
    }
}
